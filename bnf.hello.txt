<program>   := <funcdef>+
<funcdef>   := fn <ident> '(' <funcargs>  { <statlist> }'
<funcargs>  := <ident>(, <ident>)*
<statlist>  := <statement>;(<statement>)*
<statement> := <call_func>
<call_func> := <ident> '(' <string> ')'
<string>    := "<any charactor>"
<ident>     := <latter>(<latter> | <digit>)*
<latter>    := [a-zA-Z]+
<digit>     := [0 - 9]+
